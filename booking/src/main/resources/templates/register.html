註冊新帳號

帳號長度需要在3到20個字元之間
密碼至少需要6個字元
用於忘記密碼功能
返回登入
document.getElementById("registerForm").addEventListener("submit", async (e) => { e.preventDefault(); const username = document.getElementById("username").value; const password = document.getElementById("password").value; const email = document.getElementById("email").value; const fullName = document.getElementById("fullName").value; const phone = document.getElementById("phone").value; // 表單驗證 if (username.length < 3 || username.length > 20) { showMessage("帳號長度需要在3到20個字元之間", false); return; } if (password.length < 6) { showMessage("密碼至少需要6個字元", false); return; } if (!email || !email.includes('@')) { showMessage("請輸入有效的電子郵件", false); return; } const formData = new URLSearchParams(); formData.append("username", username); formData.append("password", password); formData.append("email", email); if (fullName) formData.append("fullName", fullName); if (phone) formData.append("phone", phone); try { const res = await fetch("/api/auth/register", { method: "POST", headers: {"Content-Type": "application/x-www-form-urlencoded"}, body: formData }); if (res.ok) { showMessage("註冊成功！3秒後自動跳轉到登入頁面...", true); setTimeout(() => window.location.href = "/login", 3000); } else { const errorData = await res.text(); if (res.status === 409) { showMessage("註冊失敗：此帳號已被使用", false); } else { showMessage("註冊失敗：" + errorData, false); } } } catch (error) { showMessage("註冊失敗：系統錯誤，請稍後再試", false); } }); function showMessage(message, isSuccess) { const msgDiv = document.getElementById("msg"); msgDiv.textContent = message; msgDiv.className = `alert ${isSuccess ? 'alert-success' : 'alert-danger'} mt-3`; }
